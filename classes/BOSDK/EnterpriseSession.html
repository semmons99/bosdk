    def initialize(cms, username, password, options = Hash.new)
      @session = CrystalEnterprise.getSessionMgr.logon(username, password, cms, 'secEnterprise')
      @infostore = @session.getService('', 'InfoStore')
      @connected = true
      @locale = options[:locale] || 'en_US'

      at_exit { disconnect }
    end

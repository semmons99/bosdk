    def method_missing(method, *args)
      eval "return @#{method.to_s}" if instance_variables.include?("@#{method.to_s}")
      if @obj.respond_to?(method)
        eval "return @#{method.to_s} = @obj.send(method, *args)"
      end
      super
    end
